#include <Wire.h> 
#include <LiquidCrystal_I2C.h>

// --- !!! FONTOS BE√ÅLL√çT√ÅS !!! ---
// Ide √≠rd be azt a sz√°mot, amit a Scanner tal√°lt (0x27 vagy 0x3F)!
LiquidCrystal_I2C lcd(0x27, 16, 2); 

// --- PIN BE√ÅLL√çT√ÅSOK ---
const int sensorPin = A0;    // F√©ny√©rz√©kel≈ë
const int buttonPin = 2;     // Gomb
const int buzzerPin = 8;     // Csipog√≥

// RGB LED
const int redPin = 9;
const int greenPin = 10;
const int bluePin = 11;

// --- V√ÅLTOZ√ìK ---
int lightLevel = 0;
int threshold = 400;         // K√úSZ√ñB√âRT√âK (A tegnapi be√°ll√≠t√°sod!)

int systemState = 0;         // 0=Figyel, 1=Riaszt, 2=N√©m√≠tva
int lastButtonState = LOW;
unsigned long lastBlinkTime = 0;
unsigned long lastScreenUpdate = 0;
bool ledState = false;

void setup() {
  pinMode(redPin, OUTPUT);
  pinMode(greenPin, OUTPUT);
  pinMode(bluePin, OUTPUT);
  pinMode(buzzerPin, OUTPUT);
  pinMode(buttonPin, INPUT);

  Serial.begin(9600);
  
  // LCD IND√çT√ÅSA
  lcd.init();
  lcd.backlight(); 
  
  // √údv√∂zl√©s
  lcd.setCursor(0,0); 
  lcd.print("BIZTONSAGI RDSR.");
  lcd.setCursor(0,1); 
  lcd.print("Rendszer indul..");
  
  // Teszt villan√°s
  setColor(255, 255, 255);
  delay(1500); 
  setColor(0, 0, 0);
  lcd.clear(); 
}

void loop() {
  lightLevel = analogRead(sensorPin);
  
  // GOMB KEZEL√âSE
  int currentButtonState = digitalRead(buttonPin);
  if (currentButtonState == HIGH && lastButtonState == LOW) {
    if (systemState == 1) systemState = 2;      // N√©m√≠t√°s
    else if (systemState == 2) systemState = 0; // Reset
    delay(300); 
    lcd.clear();
  }
  lastButtonState = currentButtonState;

  // LOGIKA √âS KIJELZ√âS
  
  // --- 0: FIGYEL√âS (Z√∂ld) ---
  if (systemState == 0) {
    setColor(0, 255, 0); // Z√∂ld
    noTone(buzzerPin);
    
    if (lightLevel < threshold) { // Ha er≈ës a f√©ny -> RIASZT√ÅS
      systemState = 1;
      lcd.clear();
    }
    
    // Kijelz≈ë friss√≠t√©se
    if (millis() - lastScreenUpdate > 200) {
      lastScreenUpdate = millis();
      lcd.setCursor(0,0);
      lcd.print("STATUS: FIGYEL  ");
      lcd.setCursor(0,1);
      lcd.print("Fenyero: ");
      lcd.print(lightLevel);
      lcd.print("   ");
    }
  }

  // --- 1: RIASZT√ÅS (Villog√≥ Piros) ---
  else if (systemState == 1) {
    lcd.setCursor(0,0);
    lcd.print("!!! RIASZTAS !!!");
    lcd.setCursor(0,1);
    lcd.print("BEHATOLAS!      ");

    if (millis() - lastBlinkTime > 150) {
      lastBlinkTime = millis();
      ledState = !ledState;
      if (ledState) { setColor(255, 0, 0); tone(buzzerPin, 1000); } 
      else { setColor(0, 0, 0); noTone(buzzerPin); }
    }
  }

  // --- 2: N√âM√çTVA (Folyamatos Piros) ---
  else if (systemState == 2) {
    setColor(255, 0, 0); 
    noTone(buzzerPin);
    
    lcd.setCursor(0,0);
    lcd.print("STATUS: NEMITVA ");
    lcd.setCursor(0,1);
    lcd.print("Nyomj gombot!   ");
  }
  
  delay(10);
}

void setColor(int r, int g, int b) {
  analogWrite(redPin, r);
  analogWrite(greenPin, g);
  analogWrite(bluePin, b);
}
